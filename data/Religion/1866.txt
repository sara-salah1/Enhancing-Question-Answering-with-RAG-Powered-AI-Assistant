لا يعلمها إلا الله ما العلامات التي تشير إلى قرب وقوع الساعة؟ وهل صحيح ما يردده بعض الدعاة من أن تكالب وعدوان غير المسلمين على ديار الإسلام هو من علامات قرب وقوع الساعة؟ ش . م - دبي - يقول الدكتور طه حبيشي، أستاذ العقيدة بجامعة الأزهر: ينبغي ألا نسير وراء مثل هذا الكلام الذي يشجع المسلمين على الاستسلام وعدم دفع العدوان الذي يتعرضون له، فما يحدث الآن هو عدوان ترفضه وتدينه كل الشرائع والأعراف الدولية وينبغي دفع هذا العدوان بالكفاح دفاعاً عن الدين والوطن .ومن الأمور المسلمة عند كل عاقل أن وقت قيام الساعة لا يعلمه إلا الله تعالى وحده، فهو من الغيوب التي استأثر الله تعالى بعلمها من دون أن يطلع عليه ملك مقرب، أو نبي مرسل وقد تعددت الآيات القرآنية التي أكدت هذه الحقيقة، ومنها قوله تعالى: "يسألونك عن الساعة أيان مرساها قل إنما علمها عند ربي لا يجليها لوقتها إلا هو ثقلت في السماوات والأرض لا تأتيكم إلا بغتة يسألونك كأنك حفي عنها قل إنما علمها عند الله ولكن أكثر الناس لا يعلمون" .قال العلماء: والسبب في إخفاء وقت قيام الساعة عن الناس، أن يكون المسلمون دائماً على حذر، فيكون ذلك أدعى للطاعة وأزجر عن المعصية، فإنه متى علمها المكلف فربما قصر عن التوبة وأخرها .ومع أن كل عاقل يعلم علم اليقين أن وقت قيام الساعة لا يعلمه إلا الله تعالى، إلا أنه سبحانه جعل لقرب وقت قيامها علامات معينة تدل على هذا القرب، وقد أشار سبحانه إلى ذلك في قوله تعالى: ''فهل ينظرون إلا الساعة أن تأتيهم بغتة فقد جاء أشراطها فأنى لهم إذا جاءتهم ذكراهم'' .وقد ذكر العلماء أن لقرب قيام الساعة علامات صغرى وأخرى كبرى . . أما العلامات الصغرى فمن أهمها: بعثة الرسول صلى الله عليه وسلم التي هي ختام الرسالات السماوية، ولعل مما يشير إلى ذلك قوله تعالى: "يسألونك عن الساعة أيان مرساها فيم أنت من ذكراها إلى ربك منتهاها إنما أنت منذر من يخشاها كأنهم يوم يرونها لم يلبثوا إلا عشية أو ضحاها" .وفي الحديث الصحيح عن أنس رضي الله عنه أن النبي صلى الله عليه وسلم قال: "بعثت أنا والساعة كهاتين، وأشار بالسبابة والوسطى" .والمقصود بهذا التشبيه أنه صلى الله عليه وسلم ليس بينه وبين قيام الساعة نبي آخر، فهي تليه وتأتي بعده .كذلك من علامات قرب الساعة ما جاء في الحديث الصحيح عن عمر بن الخطاب رضي الله عنه قال: بينما نحن عند رسول الله صلى الله عليه وسلم إذ طلع علينا رجل شديد بياض الثياب شديد سواد الشعر، لا يرى عليه أثر السفر ولا يعرفه منا أحد، حتى جلس إلى النبي صلى الله عليه وسلم فأسند ركبتيه إلى ركبتيه ووضع كفيه على فخذيه، وقال فيما قال: فأخبرني عن الساعة أي: عن وقت مجيئها قال صلى الله عليه وسلم "ما المسؤول عنها بأعلم من السائل . قال: فأخبرني عن أماراتها أي: علاماتها قال صلى الله عليه وسلم: أن تلد الأمة ربتها، وأن ترى الحفاة العراة العالة رعاء الشاء يتطاولون في البنيان . ." .كذلك من علامات الساعة الصغرى ما ذكره النبي صلى الله عليه وسلم في أحاديثه الصحيحة من أمور فيها ما فيها من العجائب والغرائب . . فعن أبي هريرة أن رسول الله صلى الله عليه وسلم قال: "لا تقوم الساعة حتى تقتتل فئتان عظيمتان تكون بينهما مقتلة عظيمة دعوتهما واحدة . . حتى يقبض العلم أي: يقل العلم النافع بسبب موت العلماء وتكثر الزلازل أي: كثرة زائدة على ما يعهدها الناس ويتقارب الزمان أي أن المسافات البعيدة تقطع في زمن قليل وتظهر الفتن، ويكثر الهرج أي ويكثر التقاتل بين الناس وحتى يمر الرجل بقبر الرجل فيقول: يا ليتني مكانه، أي: لما يرى من تقديم من يستحق التأخير، وتأخير من يستحق التقديم بسبب ضياع الأمانة" .وهناك علامات أخرى جاءت في بعض الأحاديث النبوية، وهي علامات المتأمل فيها يرى أن معظمها قد تحقق في أزمان متفاوتة، وأما العلامات الكبرى التي تدل على قرب قيام الساعة . . فمن أهمها: خروج الدابة التي تكلم الناس: قال تعالى: "وإذا وقع القول عليهم أخرجنا لهم دابة من الأرض تكلمهم أن الناس كانوا بآياتنا لا يوقنون" .أيضاً طلوع الشمس من مغربها من علامات قرب وقوع الساعة، حيث وردت أحاديث صحيحة صرحت بأن خروج الشمس من مغربها لا من مكان مشرقها دليل على قيام الساعة، ودليل على أنه قد أزفت الآزفة التي ليس لها من دون الله كاشفة .ومن العلامات الكبرى التي تدل على قرب قيام الساعة كما قال العلماء خروج المسيح الدجال، وقد سمي بهذا الاسم لأنه يمسح الأرض ويقطعها طولاً وعرضاً في مدة زمنية قصيرة، ولأنه أعور ممسوح العين، وعند ظهوره يدعي الألوهية، ويحاول أن يفتن الناس عن دينهم بسبب الخوارق التي يأتي بها، ثم ينتهي الأمر بقتله على يد عيسى، عليه السلام .ونزول عيسى عليه السلام من علامات قرب وقوع الساعة فقد وردت أحاديث متعددة بلغت مبلغ التواتر المعنوي تدل على أن من علامات الساعة الكبرى، نزول عيسى عليه السلام حاكماً بشريعة الإسلام التي جاء بها رسول الله صلى الله عليه وسلم وأنه يقتل المسيح الدجال . . والله أعلم . آداب إخراج الزكاة ما الآداب التي ينبغي أن يحرص عليها المسلم عند إخراجه للزكاة؟ س . ق - رأس الخيمة - تقول لجنة الفتوى بالأزهر: من آداب الزكاة أن يدفعها المزكي لغيره من المحتاجين بنية خالصة لوجه الله تعالى، وأن يتنزه عن التباهي والتفاخر، وعن الإساءة إلى الذي تقدم له الزكاة بأي لون من ألوان الإساءة لأن الله تعالى يقول: "قول معروف ومغفرة خير من صدقة يتبعها أذى" . . وأن يسلك في دفع الزكاة ما فيه الخير والمنفعة للمحتاجين، فإن وجد أن من الخير دفعها سراً، سترا لكرامة الفقراء، دفعها لهم سرا، وإن وجد أن من الخير دفعها علانية لكي يقتدي به الناس دفعها علانية ما دامت نيته طيبة وما دام بعيداً عن الرياء والتفاخر .قال تعالى: "إن تبدوا الصدقات فنعما هي وإن تخفوها وتؤتوها الفقراء فهو خير لكم" .كذلك من آداب الزكاة أن يتحرى المزكي إعطاء زكاته لمن هو أكثر احتياجا، ولمن ظاهره يدل على الاستقامة، وعلى أنه لم يلجأ إلى قبول العطاء من غيره إلا لضرورة . . فقد أمرنا الله تعالى أن نكرم الفقراء الذين هم مع فقرهم فيهم العفة . . فقال: "للفقراء الذين أحصروا في سبيل الله لا يستطيعون ضربا في الأرض يحسبهم الجاهل أغنياء من التعفف تعرفهم بسيماهم لا يسألون الناس إلحافاً وما تنفقوا من خير فإن الله به عليم . ." أي: اجعلوا زكاة أموالكم وصدقاتكم للفقراء الذين من صفاتهم العفاف والضعف عن طلب الرزق .كذلك من آداب الزكاة المبادرة بدفعها للمستحقين في وقتها من دون إبطاء أو تردد، لأن الأعمار بيد الله تعالى، ولا يدري إنسان أيعيش إلى الغد أم لا . الحلف بغير الله بعض الناس يحلفون بالطلاق أو يحلفون بآبائهم وأمهاتهم ويرفضون الحلف بالله خشية الوقوع في المحظور انطلاقا من قول الحق سبحانه ''ولا تجعلوا الله عرضة لأيمانكم'' . . فما حكم الإسلام في ذلك؟ ل . و - عجمان - يقول الدكتور أحمد عمر هاشم، أستاذ السنة النبوية، عضو هيئة كبار العلماء بالأزهر: لما كان الحلف يقتضي تعظيم المحلوف به، كانت حقيقة الحلف مختصة بالله سبحانه وتعالى، فلا يضاهى به غيره وجاء التحذير من الحلف بغير الله، حيث يقول صلى الله عليه وسلم: "من كان حالفا فليحلف بالله أو ليصمت" . . فخص الحلف بالله وحده وإلا فلا يحلف الإنسان أبدا، وهذا معنى قوله: أو "ليصمت"، والحلف بالمخلوق مكروه كالحلف بالنبي، والكعبة، وجبريل، والصحابة .كما جاء التحذير من الحلف بالآباء أو الأمهات . . ففي الحديث الصحيح: "لا تحلفوا بآبائكم ولا بأمهاتكم، ولا تحلفوا إلا بالله" . . قال الإمام الشافعي: "أخشى أن يكون الحلف بغير الله معصية"، وقوله هذا محمول على المبالغة في التنفير من ذلك فلو حلف به لم ينعقد يمينا، فإن اعتقد الحالف في المحلوف به ما يعتقده في الله كفر . . أما إذا سبق لسانه إليه بلا قصد فلا كراهة بل هو لغو يمين .
