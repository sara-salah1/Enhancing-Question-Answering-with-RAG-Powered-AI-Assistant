فاسألوا أهل الذكر إن كنتم لا تعلمونصدق الله العظيمأختي المسلمة.. أدوارك عديدة ومتشعبة، وهذه الزاوية جاءت لتنير لك الطريق عندما تصعب عليك الرؤية. موافقة الفتاة ضرورة زوجي يحاول إجبار ابنتنا على الزواج من ابن شقيقه وهي لا تريده، هل هذا من حق الأب؟ وهل يجوز أن ترتبط هي بإنسان تريده من دون موافقة والدها؟سعاد م.- أبوظبي يقول الدكتور يوسف القرضاوي: لا يجوز لأب أن يزوج ابنته ممن يريدها هو، وإن كانت هي لا تريده، بل لابد أن توافق عليه صراحة أو ضمنا، وذلك بالسكوت إذا غلبها الحياء، ولهذا قال عليه الصلاة والسلام: الأيم أحق بنفسها من وليها، والبكر تستأذن في نفسها، وإذنها صماتها.وقد روى البخاري عن خنساء بنت خدام الأنصارية أن أباها زوجها وهي ثيب فكرهت ذلك، فأتت رسول الله صلى الله عليه وسلم فرد نكاحها.وروي عن ابن عباس: أن جارية بكرا أتت النبي صلى الله عليه وسلم، فذكرت أن أباها زوجها وهي كارهة، فخيرها النبي صلى الله عليه وسلم.ولا ينبغي للآباء أن يستبدوا بزواج بناتهن، ويضربوا بعواطفهن عرض الحائط، فعن ابن عباس قال: جاء رجل إلى النبي صلى الله عليه وسلم فقال: إن عندنا يتيمة، وقد خطبها رجل معدم، ورجل موسر، وهي تهوى المعدم، ونحن نهوى الموسر. فقال صلى الله عليه وسلم: لم يُرَ للمتحابين مثل النكاح، وينبغي أن يتم الزواج برضا الأطراف كلها من أسرة المرأة، وأسرة الرجل، لهذا اشترط الفقهاء إذن ولي المرأة بل حضوره العقد، وجعل ذلك بعضهم أمرا مستحبا لا واجبا، ولهم في ذلك أدلة من القرآن والسنة، تبحث في كتب الفقه.والأولى ألا تتزوج المرأة بغير إذن وليها حتى لا يلعب بها الزوج بعد ذلك، ويضيع حقوقها ولا تجد من يدافع عنها.بل الأولى أن تشاور المرأة في زواج ابنتها، كما ورد في حديث: آمروا النساء في بناتهن، لأن الأم أعرف بابنتها من أبيها، ولأنها إذا لم توافق على الزواج فقد تنكد عيش زوج ابنتها بعد ذلك.النهي عن كفران العشير تزوجت من رجل وعشنا حياة سعيدة، ولكنه أصيب بمرض خطير ويحتاج رعاية خاصة وأنا لم أعد أتحملها، فهل لو انفصلت عنه أكون آثمة؟م. عبدالله- أم القيوين يقول الشيخ محمود عاشور الوكيل السابق للأزهر: لا شك أن الحياة متقلبة بين اليسر والعسر، والصحة والمرض، والمؤمن الصادق يثبت جدارته بالحياة في كل الأحوال كما جاء في الحديث الصحيح: عجبا لأمر المؤمن إن أمره كله له خير، وليس ذلك لأحد إلا للمؤمن، إن أصابته سراء شكر فكان خيرا له، وإن أصابته ضراء صبر فكان خيرا له.والزوجة تقر بأنها عاشت مع زوجها أياما سعيدة عندما كان صحيح الجسم وافر الثراء، فهل من المروءة والإنسانية أن تتركه في محنته لتتزوج غيره وتكمل معه مشوار حياته سعيدة كما بدأت؟إن التي تفكر في ذلك تدخل تحت حكم الحديث الذي ينهى عن كفران العشير، فقد نسيت ما قدمه لها زوجها من خير، وتبخر بسرعة ما نعمت به لسنوات طوال، وقد جاء في الحديث الصحيح أن النبي صلى الله عليه وسلم أمر النساء يوم العيد بكثرة التصدق لأن أكثرهن حطب جهنم، بسبب كثرة الشكاة وكفران العشير: لو أحسنت إلى إحداهن الدهر ثم رأت منك شيئا قالت: ما رأيت منك خيرا قط.إن الزوجة الصالحة تعين زوجها على نوائب الدهر لا أن تتخلى عنه، وكفى بالسيدة خديجة رضي الله عنها مثالا رائعا في صدق معونتها للنبي صلى الله عليه وسلم بالقول والفعل والمال، وبهاجر أم إسماعيل التي تحملت الوحدة وقاست البعد والألم طاعة لأمر الله تبارك وتعالى وأمر زوجها إبراهيم، وبأم الدحداح التي شجعت زوجها على التصدق بالحديقة في سبيل الله، وبزينب بنت النبي صلى الله عليه وسلم التي وفت لزوجها فخلصته من الأسر بأعز ما تملكه.إن النبي صلى الله عليه وسلم نهى المرأة عن أن تطلب الطلاق من زوجها إلا إذا كان هناك سبب قاهر يجعل الحياة متعذرة أو متعسرة، ففي الحديث الحسن الذي رواه الترمذي وأبو داود وابن ماجه: أيما امرأة سألت زوجها طلاقا من غير بأس فحرام عليها رائحة الجنة.الزوج المفقود هل يجوز للزوجة أن تتزوج إن وصلها خبر وفاة زوجها الذي يعمل خارج البلاد ولا تعرف مصيره؟خ.ص رأس الخيمة تقول دار الإفتاء المصرية: إن جاء خبر للزوجة أن زوجها توفي في غيابه فلا يجوز لها أن تتزوج بعده إلا بعد رفع الأمر للقاضي ليتحقق من موته، وإن حكم بموته ثم عاد قبل أن تتزوج فهي زوجته وإن عاد بعد أن تزوجت فإن كان قبل دخول الزوج الثاني بها فهي زوجة الأول، وتعود إليه بالعقد الأول ولا صداق عليه، وإن عاد بعد دخول الزوج الثاني بها خير الأول بين أخذها فتكون زوجته بالعقد الأول وبين أخذ صداقها وتكون زوجة للثاني، ويجب على الأول اعتزالها حتى تنقضي عدتها من الثاني إن اختارها زوجها الأول، وإن لم يخترها بقيت للثاني مع استئناف عقد جديد على الصحيح وإذا اختار الأول أن يتركها للثاني فيرجع بصداقها.. والله تعالى أعلم. لا ينقض الوضوء أحيانا يتساقط بعض من بول طفلي الرضيع على ملابسي هل يجب تغيير هذه الملابس وإعادة الوضوء عند الصلاة؟ه.م أم القيوين يقول د. نصر فريد واصل أستاذ الشريعة الإسلامية وعضو مجمع البحوث بالأزهر: المقرر شرعا أن بول الصبي الذي لم يأكل الطعام يكتفى في تطهيره بالرش لحديث أم قيس رضي الله عنها أنها أتت النبي صلى الله عليه وسلم بابن لها لم يبلغ أن يأكل الطعام، وأن ابنها ذاك بال في حجر النبي صلى الله عليه وسلم فدعا رسول الله صلى الله عليه وسلم بماء فنضحه على ثوبه ولم يغسله غسلا. (متفق عليه).وعن علي رضي الله عنه قال: قال رسول الله صلى الله عليه وسلم: بول الغلام ينضح عليه، وبول الجارية يغسل.ثم إن النضح إنما يجزئ مادام الصبي يقتصر على الرضاع، أما إذا أكل الطعام على جهة التغذية، فإنه يجب الغسل بلا خلاف ولعل سبب الرخصة في الاكتفاء بنضح بول الصبي هو ولوع الناس بحمله المفضي إلى كثرة بوله عليهم ومشقة غسل ثيابهم فخفف ذلك. وعلى ذلك فإن بول الصبي وإن كان الرسول صلى الله عليه وسلم قد أمر بنضح الثياب عند وقوعه لا ينقض الوضوء. النساء وزيارة القبور هل أباح الإسلام للنساء زيارة المقابر؟خ. محمد.ع- العين يقول الشيخ عبد المقصود إبراهيم من علماء الأزهر: إن النبي صلى الله عليه وسلم نهى أولا عن زيارة القبور قطعا لما كان عليه أهل الجاهلية من التفاخر بزيارتها لتعداد مآثر من فيها من الآباء والأجداد الذي يشير إليه قوله تعالى: ألهاكم التكاثر، حتى زرتم المقابر، ثم رخص لهم بعد في زيارتها لتذكر الموت والاستعداد للحياة الآخرة كما بينه الحديث الصحيح: كنت نهيتكم عن زيارة القبور فزوروا القبور فإنها تزهد في الدنيا وتذكر بالآخرة، وغير ذلك من الأحاديث التي رواها مسلم وغيره في هذا المقام.وأجمع المسلمون على استحباب زيارة القبور، وأوجبها بعض العلماء، غير أنهم قالوا: إن ذلك خاص بالرجال دون النساء، لكن لما رأى النبي صلى الله عليه وسلم ما في خروجهن من مفاسد نهاهن عنها، واستمر الإذن للرجال وقال آخرون: إن النهي عن زيارتهن كان سابقا للنهي العام عن زيارة القبور، ثم جاء الإذن للرجال، وبقي المنع مستمرا بالنسبة لهن.ومهما يكن من شيء فإن في زيارتهن أقوالا تتلخص فيما يأتي: التحريم مطلقا سواء كانت هناك فتنة أو مفسدة عند زيارتهن أم لا، ودليله حديث: لعن الله زوارات القبور.التحريم عند خوف الفتنة أو المفسدة، وبهذا يحرم على الشابات زيارة القبور وكذلك على غيرهن إذا كن بزينة أو شيء يلفت إليهن الأنظار، وتجوز للعجائز اللاتي لا يفتن بهن. الكراهة، ودليلها القياس على اتباع الجنائز، الذي ورد فيه حديث أم عطية أيضا: نهينا عن اتباع الجنائز ولم يعزم علينا.الإباحة: ودليلها عدم إنكار النبي صلى الله عليه وسلم عندما ذهبت عائشة إلى البقيع، وهي مقبرة المسلمين، وعلمها ما تقوله عند زيارة القبور وهو: السلام عليكم أهل الديار من المؤمنين والمسلمين، ويرحم الله المستقدمين منا والمستأخرين، وإنا إن شاء الله بكم لاحقون، رواه مسلم، كما أن النبي صلى الله عليه وسلم مر على امرأة تبكي عند القبر فأمرها بالتقوى والصبر، ونهاها عن البكاء لأنه سمع منها ما يكره من نواحة وغيره، ولم ينهها عن أصل الزيارة.الاستحباب: كما هي مستحبة للرجال ودليلها عموم الإذن بالزيارة في قوله صلى الله عليه وسلم: فزوروها.والآراء الثلاثة الأخيرة محلها عند أمن الفتنة والمفسدة، وإلا حرمت الزيارة، بهذا يعلم جواب السؤال، وإن كنا نميل إلى كراهية زيارتهن على الرغم من عدم وجود محرم محظور، كالسفور والنياحة واللطم والجلوس على القبر للمبيت عنده وما إلى ذلك، فإن الأولى للمرأة أن تستقر في بيتها لا تغادره إلا للضرورة أو الحاجة الملحة، صيانة لها من الفساد، والله أعلم.
