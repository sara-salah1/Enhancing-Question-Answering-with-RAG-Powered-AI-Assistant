في معرض سياق قصة قارون يوضح الله سبحانه وتعالى لنا مشهداً من الأعمال الاستفزازية التي كان قارون يقوم بها، حيث كان يسير سير الخيلاء في استعراض جذاب فتان يبرز فيه تسلطه على المال حتى يبهر العامة من الناس ومن ضعاف القلوب، فيقفون وقفة المأخوذ المبهور ويتمنون أن يرزقوا بالمال كما رزق قارون . وعلى العكس من ذلك فهناك مشهد آخر يتضمن استخفاف العلماء والعقلاء بالمظاهر المزيفة، حيث تمكنت هذه الفئة المؤمنة من الوقوف في وجه قارون بقوة إيمانها فلم تنجرف نحو عبادة المال ولم تفتتن به لاعتقادها بزوال الدنيا وأنها جسر للآخرة، وهي بمنزلة دار اختبار للوصول إلى دار القرار والعاقبة للصابرين على الحق والمتمسكين به، فالله سبحانه وتعالى يقول: فخرج على قومه في زينته، قال الذين يريدون الحياة الدنيا: يا ليت لنا مثل ما أوتي قارون أنه لذو حظ عظيم . وقال الذين أوتوا العلم ويلكم ثواب الله خير لمن آمن وعمل صالحاً ولا يلقاها إلا الصابرون . ومن الملاحظ أنه لا يقف في وجه المغريات والفتن إلا من رزق صبراً وثباتاً على الطاعات وبعداً عن المعاصي والشهوات، وما نعانيه في هذه الأيام خير دليل على ذلك . قال سبحانه وتعالى: فخسفنا به وبداره الأرض، فما كان له من فئة ينصرونه من دون الله وما كان من المنتصرين، فالعقاب أمر بديهي للمنحرف الذي لا يرجع عن غيه، ويتمادى في تمرده، وهذه الآية الكريمة تصور لنا نوع العقوبة الحاسمة القاضية على قارون وما يملك من ثروات وذلك لكثرة تماديه في الباطل ولشدة إيذائه لموسى عليه السلام، وقد ذكرت بعض التفاسير في سياق شرح هذه الآية، حادثة مزرية ومؤلمة قام بتمثيلها الضال قارون بحق موسى عليه السلام ومفادها: أنه حينما نزلت فريضة الزكاة على سيدنا موسى عليه السلام وافق قارون، بادئ الأمر، على دفع دينار عن كل ألف دينار وأن يخرج شاة عن كل ألف شاة، غير أنه تراجع عما وافق عليه لأنه سيخرج مبلغاً كبيراً كزكاة عن ماله، وجمع قومه ليحرضهم ضد موسى عليه السلام وقال لهم: إن موسى قد أمركم بكل شيء فاطعمتموه، وهو الآن يريد أن يأخذ أموالكم، فما رأيكم؟ فقالوا له: أنت كبيرنا فمرنا بما شئت . فدبر قارون مؤامرة ضد موسى عليه السلام وكان قارون يستهدف القدح بسلوك موسى عليه السلام من الناحية الأخلاقية، غير أن الله عز وجل أحاط موسى بعنايته ورحمته فكشف المؤامرة على حقيقتها قبل تنفيذها، فأعرض الناس عن قارون وحاشيته وانفضوا عنه بعد أن خاب ظنه وطاش سهمه وعندها سجد موسى لربه شاكراً وقال: اللهم إن كنت رسولك فاغضب لي، فاستجاب الله الدعاء وخسف الأرض بقارون . وحينما شعر هذا الطاغية بما حل به وبحاشيته من رفقاء السوء أخذ يتضرع إلى موسى ويناشده الصفح عنه، غير أنه لم يلتفت إلى تضرعه، ونزل عقاب الله رب العالمين المحتم الذي لا راد له .لقد كان لهول الخسف أثره القوي في قلوب الذين تمنوا أن يرزقهم الله مالاً كما أوتي قارون، وأزالت هذه العقوبة الربانية الغشاوة عن قلوبهم، والغفلة عن عقولهم، وأخذوا يحمدون الله لعدم وقوعهم في فتنة المال، وندموا على ما صدر منهم من التمنيات بعد أن رأوا الخسف ينزل بقارون وحاشيته، وعن ذلك يقول رب العالمين: (وأصبح الذين تمنوا مكانه بالأمس يقولون: ويكأن الله يبسط الرزق لمن يشاء من عباده ويقدر لولا أن منّ الله علينا لخسف بنا، ويكأنه لا يفلح الكافرون) . والمراد بلفظ (يقدر) يحبس الرزق أو يمنعه، ضد البسط . قال سبحانه وتعالى: تلك الدار الآخرة نجعلها للذين لا يريدون علواً في الأرض ولا فساداً، والعاقبة للمتقين . من جاء بالحسنة فله خير منها، ومن جاء بالسيئة فلا يجزى الذين عملوا السيئات إلا ما كانوا يعملون . فالمؤمنون الذين عزفوا عن مغريات المال وفتنته أو ثبتوا في وجه التيارات المنحرفة لا يبغون غلبة ولا تسلطاً ولا ظلماً ولا عدواناً، سيقطفون ثمار عملهم وتستقبلهم دار المقام ذات الجنات الغناء، يوم لا ينفع مال ولا بنون . إلا من أتى الله بقلب سليم (سورة الشعراء الآيتان 88 و89) . ومن رحمة الله بعباده أن الذي يقوم بعمل صالح له عشرة أمثاله من الحسنات، بينما الذي يرتكب معصية فلا يجزى إلا مثلها، ولا يظلم ربك أحداً .
