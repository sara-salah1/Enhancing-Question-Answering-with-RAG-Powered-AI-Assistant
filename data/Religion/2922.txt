آداب وأخلاق المعاملات المالية } ما الآداب والأخلاق التي ينبغي أن يحرص عليها المسلمون في معاملاتهم المالية؟ ي. ن - الفجيرة - يقول الشيخ محمود عاشور، وكيل الأزهر الأسبق: شريعة الإسلام نهت عن كل معاملة يشوبها الغش والظلم، أو الاحتكار، أو الغرر، وأمرت أتباعها بالالتزام بآداب وأخلاق راقية في معاملاتهم المالية. ومن ذلك: - ‬أمرتهم بالتيسير على المعسر،‮ ‬والرأفة به،‮ ‬وتأجيل ما عليه من ديون حتى‮ ‬يوفقه الله تعالى لسدادها‮.‬قال تعالى‮: «‬وَإِن كَانَ‮ ‬ذُو عُسْرَةٍ‮ ‬فَنَظِرَةٌ‮ ‬إلى مَيْسَرَةٍ‮ ‬وَأَن تَصَدَّقُواْ‮ ‬خَيْرٌ‮ ‬لَّكُمْ‮ ‬إِن كُنتُمْ‮ ‬تَعْلَمُونَ‮».‬ ومن الأحاديث النبوية التي‮ ‬وردت في‮ ‬ثواب الذين‮ ‬ييسرون على المعسر،‮ ‬ما روي‮ ‬عن حذيفة رضي الله عنه عن رسول الله ‬ صلى الله عليه وسلم أنه قال‮: «‬تلقت الملائكة روح رجل ممن كان قبلكم فقالوا له‮: ‬أعملت من الخير شيئا؟ قال‮: ‬لا‮. ‬قالوا‮: ‬تذكر‮. ‬قال‮: ‬كنت أداين الناس فآمر فتياني‮ ‬بأن‮ ‬ينظروا المعسر،‮ ‬ويتجوزوا عن الموسر‮.. ‬قال الله تعالى‮: «‬تجاوزوا عنه‮».‬ وأمرتهم أيضاً بتحري‮ ‬الحلال وبالبعد عن الشبهات‮.. ‬قال تعالى‮: «‬يَا أَيُّهَا النَّاسُ‮ ‬كُلُواْ‮ ‬مِمَّا فِي‮ ‬الأَرْضِ‮ ‬حَلاَلاً‮ ‬طَيِّباً‮ ‬وَلاَ‮ ‬تَتَّبِعُواْ‮ ‬خُطُوَاتِ‮ ‬الشَّيْطَانِ‮ ‬إِنَّهُ‮ ‬لَكُمْ‮ ‬عَدُوٌّ‮ ‬مُّبِينٌ».‬ ‮- ‬وأمرتهم بالحرص على العمل النافع الذي‮ ‬يعود عليهم وعلى أمتهم بالخير،‮ ‬ويكفي‮ ‬في‮ ‬هذا المعنى قوله سبحانه‮: «‬هُوَ‮ ‬الَّذِي‮ ‬جَعَلَ‮ ‬لَكُمُ‮ ‬الْأَرْضَ‮ ‬ذَلُولاً‮ ‬فَامْشُوا فِي‮ ‬مَنَاكِبِهَا وَكُلُوا مِن رِّزْقِهِ‮ ‬وَإِلَيْهِ‮ ‬النُّشُورُ‮».‬ وأمرتهم بأداء الحقوق إلى أصحابها من دون مماطلة أو تسويف‮.. ‬ففي‮ ‬الحديث الصحيح عن جابر أن رسول الله صلى الله عليه وسلم قال‮: «‬رحم الله رجلاً سمحاً إذا باع وإذا اشترى وإذا اقتضى‮».‬وفي‮ ‬حديث آخر‮: «‬من أخذ حاجة الناس‮ ‬يريد أداءها أدى الله عنه،‮ ‬ومن أخذها‮ ‬يريد إتلافها أتلفه الله».‬كما أمرتهم بأن‮ ‬يبنوا معاملاتهم على التراضي‮ ‬المشروع‮.. ‬فالله تعالى‮ ‬يقول‮: «‬يَا أَيُّهَا الَّذِينَ‮ ‬آمَنُواْ‮ ‬لاَ‮ ‬تَأْكُلُواْ‮ ‬أَمْوَالَكُمْ‮ ‬بَيْنَكُمْ‮ ‬بِالْبَاطِلِ‮ ‬إِلاَّ‮ ‬أَن تَكُونَ‮ ‬تِجَارَةً‮ ‬عَن تَرَاضٍ‮ ‬مِّنكُمْ‮».‬وأمرتهم بتوثيق المعاملات،‮ ‬لأن هذا التوثيق أدعى للاطمئنان وأصون للحقوق،‮ ‬ويكفي‮ ‬أن أطول آية في‮ ‬القرآن تحدثت عن توثيق الديون،‮ ‬هي‮ ‬قوله تعالى‮: «‬يَا أَيُّهَا الَّذِينَ‮ ‬آمَنُواْ‮ ‬إِذَا تَدَايَنتُم بِدَيْنٍ‮ ‬إلى أَجَلٍ‮ ‬مُّسَمًّى فَاكْتُبُوهُ‮....».وإذا نفذ المسلمون هذه الآداب وتلك الأوامر والنواهي‮ ‬في‮ ‬معاملاتهم،‮ ‬فتح الله عليهم بركات من السماء والأرض، والله تعالى أعلم. زواج باطل } منذ وفاة زوجي قبل ست سنوات وأنا أرفض الزواج والآن تقدم رجل للزواج مني «عرفياً» واشترط عليّ عدم إعلان هذا الزواج حفاظاً على أسرته وسمعته.. فهل هذا الزواج مباح شرعاً؟ ش. و - العين - يقول الدكتور نصر فريد واصل، مفتي مصر الأسبق، عضو هيئة كبار العلماء بالأزهر: الزواج الشرعي لا يجرى في السر ومن أهم شروط الزواج «الإشهار»، أي إعلانه للناس وما يطلبه منك هذا الرجل هو الزواج السري الذي حذر منه الإسلام. والشروط والأركان اللازمة هي: (الصيغة الشرعية الصحيحة بين الزوج أو وكيله وولي الزوجة أو وكيلها الشرعي من أحد أوليائها الشرعيين - وجود الولي الشرعي للزوجة أثناء العقد ليتولى العقد بنفسه نيابة عنها لأن الولي الشرعي ركن من أركان الزواج عند جمهور الفقهاء، وتخلف هذا الركن يبطل العقد عند الجمهور، وهذا ما نراه ملائماً لهذا الزمن منعاً للتحايل ومنعاً للفساد الذي يترتب على ما يسمى بالزواج العرفي الذي يحدث في هذه الأيام - وجود الشهود العدول عند صيغة العقد، والعدالة الظاهرة شرط في صحة شهادة الشاهدين على عقد الزواج، والعدالة الظاهرة تتطلب من الشخص ألا يكون قد ارتكب كبيرة من الكبائر، ولم يكن مصراً على ارتكاب الصغائر - إعلان النكاح - أي الزواج، عند العقد إعلاناً تاماً وذلك بالطرق المتعارف عليها بين الناس.وبناء على ذلك فإن أي عقد يصدر ولم تتوافر فيه هذه الشروط والأركان، لا يكون عقداً صحيحاً ولا يعتد به شرعاً ويكون العقد بغير ذلك في حكم الاتفاق على الزنى يوجب العقوبة التعزيزية بالنسبة للطرفين والشهود معاً.. ومن هنا فإن زواجك من هذا الرجل بهذا الشكل هو زواج سري باطل وننصحك بعدم قبوله حفاظاً على دينك وحقوقك الشرعية.. والله أعلم.
