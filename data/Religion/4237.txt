محمد نديم أحمد عبدو تختلط المفاهيم أحياناً إما بفعل فاعل أو بجهل جاهل فتثار شبهات حول بعض أحكام الإسلام بقصد التشويه لدين المحبة والسماحة والسلام.من هذه الشبهات شبهة الجزية في معاملة الآخرين الذين لم يعتنقوا هذا الدين الحنيف حيث تفرض عليهم جزية، وكذا حكم الردة في الخارجين عن هذا الدين، وكذا أحكام القصاص، وأن هذا الإكراه ينافي ويناقض التسامح الذي دعا إليه الإسلام بقوله تعالى: {ادْعُ إِلَى سَبِيلِ رَبِّكَ بِالْحِكْمَةِ وَالْمَوْعِظَةِ الْحَسَنَةِ وَجَادِلْهُمْ بِالَّتِي هِيَ أَحْسَنُ إِنَّ رَبَّكَ هُوَ أَعْلَمُ بِمَنْ ضَلَّ عَنْ سَبِيلِهِ وَهُوَ أَعْلَمُ بِالْمُهْتَدِينَ}، [النحل: 125].لكن هذه الشبهات تذوب أحيانا قبل ولادتها عند المخاض، وأحيانا تولد لكنها لا تعيش لأنها مشوهة، وأحياناً تنقلب إلى قوة وخير يظهر عظمة الإسلام ومكانته وعدله وسماحته؛ فنأخذ من هذه الشبهة قضية الجزية والردة والعياذ بالله تعالى منها. لقد فرض الله تعالى الجزية في السنة الثامنة من الهجرة بقوله تعالى: {قَاتِلُوا الَّذِينَ لَا يُؤْمِنُونَ بِاللَّهِ وَلَا بِالْيَوْمِ الْآخِرِ وَلَا يُحَرِّمُونَ مَا حَرَّمَ اللَّهُ وَرَسُولُهُ وَلَا يَدِينُونَ دِينَ الْحَقِّ مِنَ الَّذِينَ أُوتُوا الْكِتَابَ حَتَّى يُعْطُوا الْجِزْيَةَ عَنْ يَدٍ وَهُمْ صَاغِرُونَ}، [التوبة: 29]، من هذا النص القرآني العظيم يتضح أن الجزية ثابتة نصاً بالقرآن لا يمكن إنكارها؛ وهي مال يعطى للحامي جزاء الحماية فإن عجز الحامي عن الحماية لم يستحقه فهي تسقط بالعجز عن الحماية لمن أخذت منه وترد عليه وعليه فإنها ليست عقوبة ولا إذلالاً بل هي أجرة لقاء عمل.لقد عرفها العلماء فقالوا: هي مال يدفعه أهل الكتاب إلى المسلمين وفي ذلك يقول الإمام السيوطي رحمه الله تعالى في «معجم مقاليد العلوم في الحدود والرسوم»: الجِزْيَة: مَا يقْضى من المَال على أهل الكتاب.ا. ه، لكن قد يستشكل لفظ الصغار على البعض، وإن فيه إهانة، لكن الحقيقة غير ذلك كما جاء في كتاب «أحكام أهل الذمة» لابن القيم رحمه الله تعالى ما نصه: اختلف الناس في تفسير (الصغار) الذي يكونون عليه وقت أداء الجزية فقال عكرمة: أن يدفعها وهو قائم، ويكون الآخذ جالساً، وقالت طائفة: أن يأتي بها بنفسه ماشياً لا راكباً.. وهذا كله مما لا دليل عليه، ولا هو مقتضى الآية ولا نقل عن رسول الله - صلى الله عليه وسلم - ولا عن الصحابة أنهم فعلوا ذلك. والصواب في الآية أن الصغار هو التزامهم لجريان أحكام الملة عليهم، وإعطاء الجزية.ا.هالجزية والإكراه والتسامحيظن البعض أن الجزية هي أخذ للمال بغير وجه حق وأنها بسبب عدم الانقياد للإسلام وهذا غير صحيح لأن أخذ الجزية في مقابل نفقات الدفاع والتكفل بحماية من دفعها ولا يكلف الدافع بالخدمة الإلزامية كما يقال: (الجهاد) لكن المسلمين مكلفون بالدفاع عنه في حال تعرضه لأي خطر أو أي سلب لحريته، بل تسقط الجزية في حال عدم القدرة على الحماية كما تقدم ولقد جاء في كتاب الخراج لأبي يوسف يعقوب بن إبراهيم بن حبيب بن سعد بن حبتة الأنصاري (المتوفى: 182ه) في قضية فتح أبي عبيدة بن الجراح رضي الله عنه الشام وأخذه الجزية من أهلها الذين كانوا يومئذ لا يزالون على دينهم واشترطوا عليه أن يحميهم من الروم الذين كانوا يسومونهم سوء العذاب وقبل أبو عبيدة الشرط لكن هرقل أعد جيشاً عظيماً لاسترداد الشام من المسلمين وبلغت الأنباء أبا عبيدة فكتب إلى عماله برد الجزية إلى الناس وهذا نص القصة: فكتب أبوعبيدة إلى كل والٍ ممن خلفه في المدن التي صالح أهلها يأمرهم أن يردوا عليهم ما جبي منهم من الجزية والخراج، وكتب إليهم أن يقولوا لهم: إنما رددنا عليكم أموالكم؛ لأنه قد بلغنا ما جمع لنا من الجموع، وأنكم اشترطتم علينا أن نمنعكم، وإنا لا نقدر على ذلك، وقد رددنا عليكم ما أخذنا منكم ونحن لكم على الشرط وما كتبنا بيننا وبينكم إن نصرنا الله عليهم؛ فلما قالوا ذلك لهم، وردوا عليهم الأموال التي جبوها منهم، قالوا: ردكم الله علينا ونصركم عليهم، فلو كانوا هم لم يردوا علينا شيئا وأخذوا كل شيء بقي لنا حتى لا يدعوا لنا شيئا.اه، فأين الإكراه المناقض للتسامح؟ من يأخذ مالاً ويرده إلى أهله وقت استنفار وحاجة إلى المال! وهذا غاية في تبيين مقصد الجزية وغاية في التسامح الذي يوجب رد الحقوق إلى أهلها حيث رددنا له المال عندما شعرنا بالعجز عن الحماية.إن هذه الشبهة بائدة يثبت ذلك غير المسلمين أيضاً، جاء في كتاب الحضارة الإسلامية للمؤرخ الغربي «آدم ميتز»: كان أهل الذمة - بحكم ما يتمتَّعون به من تسامُح المسلمين معهم، ومن حمايتهم لهم يدفعون الجزية كل منهم بحسب قُدرته، وكانت هذه الجزية أشبه بضريبة الدفاع الوطني، فكان لا يدفعها إلا الرجل القادر على حملِ السلاح؛ فلا يدفعها ذوو العاهات.. ولا المترهِّبون وأهل الصوامع إلّا إذا كان لهم يَسارٌ.ا.ه بتصرف.إن الحماية التي كانت في مقابل الجزية لم يكن فيها أي استغلال أو إذلال أو نهب لخيرات الناس آنذاك بل كانت مبلغاً رمزياً جداً يتناسب مع مكانة الدولة الرحيمة وقيمها وأخلاقها في التعامل مع الذين لم يرضوا اعتناق دينها والتي ردت حرية الاختيار في اعتناق الدين إلى الإنسان؛ والتاريخ مملوء بما يدل على ذلك لمن أراد البحث والتأكد فكم من أناس بقوا على أديانهم لم يمسوا بأذى في أنفسهم أو معابدهم أو أهليهم، بل لقد رُدّت الجزية على من أخذت منه أحياناً تكفلاً به إذا أصابه مرض أو عجز أو كبر.تاريخ الجزيةيظن البعض أن الإسلام هو أول من فرض الجزية والحقيقة خلاف ذلك يقول جورجي زيدان في كتابه «التمدن الإسلامي»: إن الجزية ليست من محدثات الإسلام بل هي قديمة من أول عهد التمدن القديم وقد وضعها يونان أثينا على سكان سواحل آسيا الصغرى حوالي القرن الخامس قبل الميلاد مقابل حمايتهم من هجمات الفينيقيين وفينيقيا يومئذ كانت تابعة لبلاد الفرس فهان على سكان تلك السواحل دفع المال في مقابل تلك الحماية والرومان وضعوا الجزية على الأمم التي أخضعوها لحكمهم وكان الفرس أيضًا يجبون الجزية من رعاياهم.اه (الجزء الأول صفحة 169 وما بعدها، للأستاذ جورجي زيدان، طبعة 1902 مطبعة الهلال مصر).وليس المراد من البحث في تاريخ الجزية تبريرها؛ بل المراد دفع الشبهة في أن الإسلام أول من فرضها، لأن تاريخها فيمن أوجبها قبل الإسلام ومن استلبها من أهلها بعد الإسلام حافل بالاستغلال التام لمن دفعها في سبيل الإبقاء على حياته ليقوم بالعمل والدفع للحامي له.حكم الردةالردة هي الخروج من عقيدة الإسلام بالقول أو بالفعل لكن لا يعد حكم الردة وما يترتب عليه إكراها منافيا للتسامح ولحرية الإنسان في الاختيار وذلك لأن الإكراه المنفي هو الإكراه على الدين عقيدة وإيماناً يقول الله تعالى: {لَا إِكْرَاهَ فِي الدِّينِ قَدْ تَبَيَّنَ الرُّشْدُ مِنَ الْغَيِّ فَمَنْ يَكْفُرْ بِالطَّاغُوتِ وَيُؤْمِنْ بِاللَّهِ فَقَدِ اسْتَمْسَكَ بِالْعُرْوَةِ الْوُثْقَى لَا انْفِصَامَ لَهَا وَاللَّهُ سَمِيعٌ عَلِيمٌ} [البقرة: 256].ومعنى هذه الآية أن الله لم يكره خلقه وهو خالقهم وكان من الممكن أن يقهر الله الناس كما قهر السماوات والأرض والحيوان والنبات والجماد ولو أكرههم لما أرسل الرسل ولذلك يقول المولى عز وجل: {وَلَوْ شَآءَ رَبُّكَ لآمَنَ مَن فِي الأرض كُلُّهُمْ جَمِيعاً أَفَأَنتَ تُكْرِهُ الناس حتى يَكُونُواْ مُؤْمِنِينَ} [ يونس: 99 ]، جاء في كتاب التحرير والتنوير للعلامة الشيخ محمد الطاهر بن عاشور التونسي رحمه الله تعالى: (ونفي الإكراه خبر في معنى النهي والمراد نفي أسباب الإكراه في حُكم الإسلام أي لا تكرهوا أحداً على أتباع الإسلام قسراً.ه).لكن من دخل دائرة التكليف صار مكلفاً بأحكامه حلاً وحرمة وفاء منه بالعهد الذي اختار الموافقة عليه.إن عقوبة الردة إمعان في حرية الاعتقاد وإعمال لأحكام الإسلام وضبط للتلاعب في الأحكام فهو حكم استقرار لا حكم إجبار أو اختيار بل إن من حكمة الإسلام أن يعلن حكم الردة لمن أراد أنْ يؤمن قبل أن يدخل الإسلام ليكون على بصيرة من أمره فمَنْ أعلن أنه مسلم وشهد أن لا إله إلا الله وأن محمداً صلى الله عليه وسلم رسول الله التزم بكل أحكام الإسلام وهو محسوب على الإسلام حقا ومطالبة ودفاعاً فإنْ أخلْ بحكمٍ من أحكام الإسلام فلا بد من محاسبته وإن الإكراه على البقاء ليس كالإكراه على الابتداء؛ ولو خلطنا بين مفهوم عدم الإكراه والتسامح غير المنضبط وحرية الفرد لكانتْ فروضُ الدين وأركانه عرضةً للتحلُّل منها دونَ زاجِر كالصلاة والصوم والزكاة وأصبحَ المسلمُ وغيرُه في حكم واحد.إن حكم الردة وما يترتب عليه يؤخذ في سياقته فلا يدخل في الإكراه المنفي ولا التسامح غير المنضبط! فهل عندما ينفذ حكم إعدام في مجرم يعد ذلك منافياً للتسامح أم حفاظاً على بقية الأرواح والأنفس؟إن للحرية والتسامح ضوابط يجب أن تراعى وإلا سادت الفوضى وضاعت القيم والأخلاق إذ لا يمكن أن يكون إعطاء الحرية للإخلال بالعهود تسامحا بل إن التسامح هو الإكراه على الوفاء بالعهود خاصة إذا كان الإخلال متسبباً في ضرر خاص أو عام فمن العبث في مثل هذه الصورة أن يترك الشخص هملاً طليقاً تحت ذريعة ممارسة الحرية وعدم الإكراه بل يجب أن يوقف بأي وسيلة حتى ولو كانت الوسيلة القتل إن لم ينته عما هو فيه من ضرر عام أو خاص.نقاش فقهيلكن يبقى إثبات حكم الردة وما يترتب عليه من ذلك كطلب الاستتابة أو القتل أو الترك إن كان ثمة مصلحة أو عذر يبقى هذا كله في دائرته الفقهية لذوي التخصص، وهو أمر من السياسة الشرعية المناطة بالأحكام السلطانية والتي لا يحكم بها إلّا السلطان من دون - إنكار لحكم الردة - وإنما وضعه بنصابه الشرعي وتنفيذه بشروطه ومراعة ظروفه.حيث إن هناك من قال إن الردة لا تستوجب القتل مستدلا بأن هناك من ارتد في عصر رسول الله صلى الله عليه وسلم ولم يقتله وقد جاء ذلك في قوله تعالى {إِنَّ الَّذِينَ آمَنُوا ثُمَّ كَفَرُوا ثُمَّ آمَنُوا ثُمَّ كَفَرُوا ثُمَّ ازْدَادُوا كُفْرًا لَمْ يَكُنِ اللَّهُ لِيَغْفِرَ لَهُمْ وَلَا لِيَهْدِيَهُمْ سَبِيلًا} [النساء: 137].نعم إن جدلا قد وجد بل قد دار في حكم قتل المرتد الثابت فقهياً وتاريخياً ومتى يحكم على الشخص بأنه مرتد؟ ومتى يقتل؟ إلى غير ذلك من بحث حكم الردة فقهياً وهو نقاش معتبر لمن وقف عنده حتى وصل الأمر إلى عدم التسليم بعموم قتل المرتد وتعليل قتال الخليفة أبي بكر رضي الله عنه للمرتدين
